Pre-process repo descriptions:¶

Specifically, place all steps into a decorated custom pipeline

  • keep only repos with descriptions <= 280 characters
  • keep only repos where we are confident that the language is English
  • remove "Awesome" lists
  • remove stop words, punctuation, and then normalize text

I'll print the cleaned data too 👇

In [11]:
@Language.component("clean_text")
def clean_text(doc):

    vocab=doc.vocab

    if len(doc.text) <= 280 \
        and doc._.language == 'en' \
        and doc._.language_score >=.8 \
        and not 'list' in doc.text.lower():

        text_list=[token.lemma_ for token in doc if not token.is_stop and not token.is_punct]
    
        return Doc(vocab, text_list)

dictionary = corpora.Dictionary(d.split() for d in docs)
bow = [dictionary.doc2bow(d.split()) for d in docs]
lda = gensim.models.ldamodel.LdaModel
num_topics=3
ldamodel = lda(bow,num_topics=num_topics,id2word=dictionary,passes=50,minimum_probability=0)
